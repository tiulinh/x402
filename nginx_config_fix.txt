server {
    server_name zorro.team www.zorro.team;

    location / {
        proxy_pass http://localhost:4021;
        proxy_http_version 1.1;

        # Essential headers
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # ⬇️ THÊM: Headers cho x402/CDP facilitator
        proxy_set_header X-Payment $http_x_payment;
        proxy_set_header X-Payer-Address $http_x_payer_address;
        proxy_set_header Authorization $http_authorization;
        proxy_pass_request_headers on;

        proxy_cache_bypass $http_upgrade;
    }

    listen 443 ssl http2;
    ssl_certificate /etc/letsencrypt/live/zorro.team/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/zorro.team/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    if ($host = www.zorro.team) {
        return 301 https://$host$request_uri;
    }

    if ($host = zorro.team) {
        return 301 https://$host$request_uri;
    }

    listen 80;
    server_name zorro.team www.zorro.team;
    return 404;
}
